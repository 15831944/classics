\newcommand\glossdisplay[4]{%
    \ifthenelse{\equal{\glstype}{common}}{%
        % common words just get printed
        #1%
    }{%
        \ifthenelse{\glsentrycurrcount{\glslabel} = 0}{%
            % this is the first time we've seen this word so add a footnote
            #1\let\thefootnote\relax\footnote{\textbf{\glstext} #2}%
        }{%
            % this isn't the first time we've seen this word so just print it
            #1%
        }%
    }%
}
